**Step 2 — Define `ScreenTimeReasonCard` (private struct).**

```swift
private struct ScreenTimeReasonCard: View {
    let eyebrow: String          // "01", "02", "03"
    let headline: String         // "Make the math personal", etc.
    let body: String?            // optional supporting line if existing copy has one

    var body: some View {
        HStack(alignment: .top, spacing: 14) {
            Text(eyebrow)
                .font(.system(size: 13, weight: .heavy, design: .monospaced))
                .tracking(1.2)
                .foregroundStyle(OB.fg3)
                .frame(width: 32, alignment: .leading)
                .padding(.top, 2)

            VStack(alignment: .leading, spacing: 4) {
                Text(headline)
                    .font(.brand(size: 17, weight: .heavy))
                    .foregroundStyle(OB.fg)
                    .fixedSize(horizontal: false, vertical: true)

                if let body, !body.isEmpty {
                    Text(body)
                        .font(.brand(size: 14, weight: .medium))
                        .foregroundStyle(OB.fg2)
                        .fixedSize(horizontal: false, vertical: true)
                }
            }
            .frame(maxWidth: .infinity, alignment: .leading)
        }
        .padding(.horizontal, 18)
        .padding(.vertical, 14)
        .frame(maxWidth: .infinity, alignment: .leading)
        .background(
            RoundedRectangle(cornerRadius: 16, style: .continuous)
                .fill(OB.surface)
        )
        .overlay(
            RoundedRectangle(cornerRadius: 16, style: .continuous)
                .stroke(OB.border, lineWidth: 1)
        )
    }
}
```

This pattern is the warmer OB-system reason-row treatment. Numbered eyebrow (mono, tracking) + bold headline + optional dim body line. Cards are stacked with 12pt vertical gap.
