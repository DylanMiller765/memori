**Step 3 — Add the tap handlers + entrance animation as `private func`s on `OnboardingView`:**

```swift
@State private var counterPulse: Bool = false
@State private var counterCoralFlash: Bool = false
@State private var cardsAppeared: [Bool] = Array(repeating: false, count: 6)

private func handleGoalTap(goal: UserFocusGoal) {
    guard let user = users.first else { return }
    let isCurrentlySelected = user.focusGoals.contains(goal)
    let currentCount = user.focusGoals.count

    if isCurrentlySelected {
        // Deselect — always allowed
        user.focusGoals.remove(goal)
        UIImpactFeedbackGenerator(style: .light).impactOccurred()
    } else if currentCount < 3 {
        // Select — under cap
        user.focusGoals.insert(goal)
        UIImpactFeedbackGenerator(style: .light).impactOccurred()
        // Counter pulse
        withAnimation(.spring(response: 0.18, dampingFraction: 0.70)) { counterPulse = true }
        DispatchQueue.main.asyncAfter(deadline: .now() + 0.18) {
            withAnimation(.easeOut(duration: 0.10)) { counterPulse = false }
        }
    } else {
        // Cap reached — pulse-only-no-replacement (UI-SPEC §"Empty / Error / Edge-case states")
        UIImpactFeedbackGenerator(style: .light).impactOccurred()
        withAnimation(.spring(response: 0.20, dampingFraction: 0.70)) {
            counterPulse = true
            counterCoralFlash = true
        }
        DispatchQueue.main.asyncAfter(deadline: .now() + 0.20) {
            withAnimation(.easeOut(duration: 0.18)) {
                counterPulse = false
                counterCoralFlash = false
            }
        }
    }
}

private func handleGoalsContinue() {
    guard let user = users.first else { return }
    let count = user.focusGoals.count

    if count == 0 {
        // Disabled-state tap — light haptic, no advance
        UIImpactFeedbackGenerator(style: .light).impactOccurred()
        return
    }

    UIImpactFeedbackGenerator(style: .medium).impactOccurred()
    Analytics.onboardingStep(step: "goals")
    withAnimation { currentPage = 6 } // → empathy
}

private func animateGoalsEntrance() {
    cardsAppeared = Array(repeating: false, count: 6)
    let reduceMotion = UIAccessibility.isReduceMotionEnabled
    let stagger = reduceMotion ? 0.0 : 0.04

    for i in 0..<6 {
        DispatchQueue.main.asyncAfter(deadline: .now() + 0.32 + Double(i) * stagger) {
            withAnimation(reduceMotion ? .easeOut(duration: 0.18) : .easeOut(duration: 0.30)) {
                cardsAppeared[i] = true
            }
        }
    }
}
```
