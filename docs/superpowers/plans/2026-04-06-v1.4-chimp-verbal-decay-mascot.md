**Files:**

The Chimp Test: Numbers (1-N) appear on a grid. They briefly show, then hide behind squares. Tap them in ascending order (1, 2, 3...). Each level adds one more number. Three strikes and you're out.

- [ ] **Step 1: Create `ChimpTestView.swift` with ViewModel**

//
//  ChimpTestView.swift

import SwiftUI
import SwiftData

// MARK: - ViewModel

@MainActor @Observable
final class ChimpTestViewModel {
    enum Phase: Equatable { case setup, playing, finished }
    enum RoundPhase: Equatable { case showing, hidden }

    var phase: Phase = .setup
    var roundPhase: RoundPhase = .showing
    var startTime: Date?
    var level = 4 // Start with 4 numbers
    var lives = 3
    var bestLevel = 0
    var nextExpectedNumber = 1

    // Grid: 8 columns x 6 rows = 48 cells
    let columns = 8
    let rows = 6
    var numberPositions: [(number: Int, row: Int, col: Int)] = []
    var revealedNumbers: Set<Int> = [] // Numbers already tapped correctly
    var wrongCell: (row: Int, col: Int)? = nil

    // Challenge support
    var challengeSeed: Int?
    private var rng: SeededGenerator?

    var durationSeconds: Int {
        guard let start = startTime else { return 0 }
        return Int(Date.now.timeIntervalSince(start))
    }

    var score: Double {
        // Normalize: level 4 = 0.0, level 12+ = 1.0
        return min(1.0, max(0.0, Double(bestLevel - 4) / 8.0))
    }

    var leaderboardScore: Int {
        // Primary: best level, secondary: time bonus
        bestLevel * 1000 + max(0, 999 - durationSeconds)
    }

    var difficulty: Int {
        // Map level to 1-10 difficulty
        min(10, max(1, bestLevel - 3))
    }

    var ratingText: String {
        switch bestLevel {
        case 0...4: return "Getting Started"
        case 5: return "Not Bad"
        case 6: return "Average"
        case 7: return "Good"
        case 8: return "Great"
        case 9: return "Excellent"
        case 10: return "Outstanding"
        case 11: return "Exceptional"
        default: return "Superhuman"
        }
    }

    func startGame() {
        phase = .playing
        startTime = Date()
        level = 4
        lives = 3
        bestLevel = 0

        if let seed = challengeSeed {
            rng = SeededGenerator(seed: UInt64(seed))
        }

        generateRound()
    }

    func generateRound() {
        revealedNumbers = []
        nextExpectedNumber = 1
        wrongCell = nil
        roundPhase = .showing

        // Generate random positions for numbers 1...level
        var positions: [(number: Int, row: Int, col: Int)] = []
        var usedPositions: Set<String> = []

        for num in 1...level {
            var row: Int, col: Int
            repeat {
                if var r = rng {
                    row = Int.random(in: 0..<rows, using: &r)
                    col = Int.random(in: 0..<columns, using: &r)
                    rng = r
                } else {
                    row = Int.random(in: 0..<rows)
                    col = Int.random(in: 0..<columns)
                }
            } while usedPositions.contains("\(row),\(col)")
            usedPositions.insert("\(row),\(col)")
            positions.append((number: num, row: row, col: col))
        }

        numberPositions = positions
    }

    func tapCell(row: Int, col: Int) {
        guard phase == .playing else { return }

        // Find if there's a number at this position
        guard let tapped = numberPositions.first(where: { $0.row == row && $0.col == col }) else {
            return // Empty cell, ignore
        }

        // Already revealed
        if revealedNumbers.contains(tapped.number) { return }

        if tapped.number == nextExpectedNumber {
            // Correct!
            revealedNumbers.insert(tapped.number)

            // After tapping 1, hide all remaining numbers
            if nextExpectedNumber == 1 {
                withAnimation(.easeOut(duration: 0.15)) {
                    roundPhase = .hidden
                }
            }

            nextExpectedNumber += 1

            // Completed the level
            if nextExpectedNumber > level {
                bestLevel = max(bestLevel, level)
                level += 1
                HapticService.shared.correct()
                // Brief delay then next round
                DispatchQueue.main.asyncAfter(deadline: .now() + 0.6) { [self] in
                    generateRound()
                }
            } else {
                HapticService.shared.tap()
            }
        } else {
            // Wrong!
            wrongCell = (row: row, col: col)
            lives -= 1
            HapticService.shared.wrong()
            bestLevel = max(bestLevel, level - 1)

            if lives <= 0 {
                // Game over
                withAnimation(.easeInOut(duration: 0.3)) {
                    phase = .finished
                }
            } else {
                // Show correct positions briefly, then retry same level
                withAnimation(.easeOut(duration: 0.15)) {
                    roundPhase = .showing
                }
                DispatchQueue.main.asyncAfter(deadline: .now() + 1.0) { [self] in
                    generateRound()
                }
            }
        }
    }

    func reset() {
        phase = .setup
        roundPhase = .showing
        startTime = nil
        level = 4
        lives = 3
        bestLevel = 0
        nextExpectedNumber = 1
        numberPositions = []
        revealedNumbers = []
        wrongCell = nil
    }
}

struct ChimpTestView: View {
    @Environment(\.modelContext) private var modelContext
    @Environment(\.dismiss) private var dismiss
    @Environment(AchievementService.self) private var achievementService

    @Environment(StoreService.self) private var storeService

    @State private var shareImage: UIImage?
    @State private var showingShareSheet = false
    @State private var showingPaywall = false
    @State private var isNewPersonalBest = false
    @State private var exerciseSaved = false

    private var user: User? { users.first }

    var body: some View {
        VStack(spacing: 0) {
            switch viewModel.phase {
            case .setup:
                setupView
            case .playing:
                playingView
            case .finished:
                resultsView

        .background(AppColors.primaryBackground)
        .navigationBarBackButtonHidden(viewModel.phase == .playing)
        .animation(.easeInOut(duration: 0.3), value: viewModel.phase)

                .font(.body)
                .foregroundStyle(AppColors.secondaryText)
                .multilineTextAlignment(.center)
                .padding(.horizontal, 32)

            VStack(spacing: 8) {
                Label("Numbers hide after you tap the first one", systemImage: "eye.slash")
                Label("Each level adds one more number", systemImage: "plus.circle")
                Label("3 lives — don't tap the wrong square!", systemImage: "heart.fill")

            .font(.subheadline)
            .foregroundStyle(AppColors.secondaryText)

            Text("Can you beat a chimpanzee?")
                .font(.headline)
                .foregroundStyle(AppColors.amber)
                .padding(.top, 8)

            Spacer()

            Button {
                viewModel.startGame()
            } label: {
                Text("Start")
                    .font(.headline)
                    .foregroundStyle(.white)
                    .frame(maxWidth: .infinity)
                    .padding(.vertical, 16)
                    .background(AppColors.amber, in: RoundedRectangle(cornerRadius: 16))
            }
            .padding(.horizontal, 24)
            .padding(.bottom, 32)
        }
    }

    // MARK: - Playing

    private var playingView: some View {
        VStack(spacing: 12) {
            // Header: level + lives
            HStack {
                Text("Level \(viewModel.level)")
                    .font(.title2.bold())
                    .foregroundStyle(AppColors.primaryText)

                Spacer()

                HStack(spacing: 4) {
                    ForEach(0..<3, id: \.self) { i in
                        Image(systemName: i < viewModel.lives ? "heart.fill" : "heart")
                            .foregroundStyle(i < viewModel.lives ? AppColors.coral : AppColors.secondaryText.opacity(0.3))
                            .font(.title3)
                    }
                }
            }
            .padding(.horizontal, 16)
            .padding(.top, 8)

            // Grid
            let cellSize = (UIScreen.main.bounds.width - 32 - CGFloat(viewModel.columns - 1) * 4) / CGFloat(viewModel.columns)

            VStack(spacing: 4) {
                ForEach(0..<viewModel.rows, id: \.self) { row in
                    HStack(spacing: 4) {
                        ForEach(0..<viewModel.columns, id: \.self) { col in
                            cellView(row: row, col: col, size: cellSize)
                        }
                    }
                }
            }
            .padding(.horizontal, 16)

            Spacer()
        }
    }

    @ViewBuilder
    private func cellView(row: Int, col: Int, size: CGFloat) -> some View {
        let numberAtCell = viewModel.numberPositions.first(where: { $0.row == row && $0.col == col })
        let isRevealed = numberAtCell.map { viewModel.revealedNumbers.contains($0.number) } ?? false
        let isWrong = viewModel.wrongCell.map { $0.row == row && $0.col == col } ?? false

        Button {
            viewModel.tapCell(row: row, col: col)
        } label: {
            ZStack {
                if let num = numberAtCell, !isRevealed {
                    if viewModel.roundPhase == .showing {
                        // Show the number
                        RoundedRectangle(cornerRadius: 6)
                            .fill(AppColors.amber)
                        Text("\(num.number)")
                            .font(.system(size: size * 0.45, weight: .bold, design: .rounded))
                            .foregroundStyle(.white)
                    } else {
                        // Hidden — show blank square
                        RoundedRectangle(cornerRadius: 6)
                            .fill(isWrong ? AppColors.coral : AppColors.accent)
                    }
                } else if isRevealed {
                    // Already tapped correctly — show green briefly then empty
                    RoundedRectangle(cornerRadius: 6)
                        .fill(AppColors.teal.opacity(0.3))
                } else {
                    // Empty cell
                    RoundedRectangle(cornerRadius: 6)
                        .fill(AppColors.secondaryText.opacity(0.08))
                }
            }
        }
        .frame(width: size, height: size)
        .disabled(numberAtCell == nil || isRevealed)
    }

    // MARK: - Results

    private var resultsView: some View {
        ScrollView {
            VStack(spacing: 24) {
                Spacer().frame(height: 20)

                if isNewPersonalBest {
                    Label("New Personal Best!", systemImage: "trophy.fill")

                VStack(spacing: 4) {
                    Text("\(viewModel.bestLevel)")
                        .font(.system(size: 72, weight: .bold, design: .rounded))
                        .foregroundStyle(AppColors.amber)
                    Text("NUMBERS")
                        .font(.caption.bold())
                        .foregroundStyle(AppColors.secondaryText)
                        .tracking(2)

                Text(viewModel.ratingText)
                    .font(.title3.bold())
                    .foregroundStyle(AppColors.accent)

                // Stats
                HStack(spacing: 32) {
                    VStack(spacing: 4) {
                        Text("\(viewModel.durationSeconds)s")
                            .font(.title3.bold())
                            .foregroundStyle(AppColors.primaryText)
                        Text("Time")
                            .font(.caption)
                            .foregroundStyle(AppColors.secondaryText)
                    }
                    VStack(spacing: 4) {
                        Text(String(format: "%.0f%%", viewModel.score * 100))
                            .font(.title3.bold())
                            .foregroundStyle(AppColors.primaryText)
                        Text("Score")
                            .font(.caption)
                            .foregroundStyle(AppColors.secondaryText)
                    }
                }
                .padding()
                .background(.ultraThinMaterial, in: RoundedRectangle(cornerRadius: 16))

                // Action buttons
                VStack(spacing: 12) {
                    Button {
                        exerciseSaved = false
                        viewModel.reset()
                    } label: {
                        Label("Try Again", systemImage: "arrow.counterclockwise")
                            .font(.headline)
                            .foregroundStyle(.white)
                            .frame(maxWidth: .infinity)
                            .padding(.vertical, 16)
                            .background(AppColors.amber, in: RoundedRectangle(cornerRadius: 16))
                    }

                    Button {
                        generateShareCard()
                        showingShareSheet = true
                    } label: {
                        Label("Share", systemImage: "square.and.arrow.up")
                            .font(.headline)
                            .foregroundStyle(AppColors.accent)
                            .frame(maxWidth: .infinity)
                            .padding(.vertical, 16)
                            .background(.ultraThinMaterial, in: RoundedRectangle(cornerRadius: 16))

                }
                .padding(.horizontal, 24)

                Button("Done") { dismiss() }
                    .font(.subheadline)
                    .foregroundStyle(AppColors.secondaryText)
                    .padding(.bottom, 32)
            }
        }
        .onAppear { saveExercise() }
        .sheet(isPresented: $showingShareSheet) {
            if let image = shareImage {
                ShareSheetView(items: [image])
            }
        }
    }

    // MARK: - Save & Share

    private func saveExercise() {
        guard !exerciseSaved else { return }
        exerciseSaved = true
        paywallTrigger.recordExerciseCompleted()
        trainingManager.addTrainingTime(viewModel.durationSeconds)

        let exercise = Exercise(
            type: .chimpTest,
            difficulty: viewModel.difficulty,
            score: viewModel.score,
            durationSeconds: viewModel.durationSeconds
        )
        modelContext.insert(exercise)

        let descriptor = FetchDescriptor<DailySession>(sortBy: [SortDescriptor(\.date, order: .reverse)])
        let allSessions = (try? modelContext.fetch(descriptor)) ?? []
        let session: DailySession
        if let existing = allSessions.first(where: { Calendar.current.isDateInToday($0.date) }) {
            session = existing
        } else {
            session = DailySession()
            modelContext.insert(session)
        }
        session.addExercise(exercise)
        user?.updateStreak()

        isNewPersonalBest = PersonalBestTracker.shared.record(score: viewModel.leaderboardScore, for: .chimpTest)

        if let user {
            _ = ContentView.awardXP(
                user: user,
                score: viewModel.score,
                difficulty: viewModel.difficulty,
                achievementService: achievementService,
                modelContext: modelContext,
                gameCenterService: gameCenterService,
                exerciseType: .chimpTest,
                gameScore: viewModel.bestLevel
            )
        }

        gameCenterService.reportScore(viewModel.leaderboardScore, leaderboardID: GameCenterService.chimpTestLeaderboard)
    }

    private func generateShareCard() {
        let card = ExerciseShareCard(
            exerciseName: "Chimp Test",
            exerciseIcon: "pawprint.fill",
            accentColor: AppColors.amber,
            mainValue: "\(viewModel.bestLevel)",
            mainLabel: "NUMBERS REMEMBERED",
            ratingText: viewModel.ratingText,
            stats: [
                ("Time", "\(viewModel.durationSeconds)s"),
                ("Score", String(format: "%.0f%%", viewModel.score * 100))
            ]
        )
        shareImage = card.renderAsImage(size: CGSize(width: 360, height: 640), scale: 3)
    }
}

// MARK: - ViewModel

@MainActor @Observable
final class VerbalMemoryViewModel {
    enum Phase: Equatable { case setup, playing, finished }

    var phase: Phase = .setup
    var startTime: Date?
    var lives = 3
    var streak = 0
    var bestStreak = 0
    var totalSeen = 0
    var totalCorrect = 0

    var currentWord = ""
    private var seenWords: Set<String> = []
    private var wordPool: [String] = []
    private var currentWordIsNew = true
    private var showingFeedback = false

    var challengeSeed: Int?
    private var rng: SeededGenerator?

    var durationSeconds: Int {
        guard let start = startTime else { return 0 }
        return Int(Date.now.timeIntervalSince(start))
    }

    var score: Double {
        // Normalize: streak 0 = 0.0, streak 50+ = 1.0
        return min(1.0, max(0.0, Double(bestStreak) / 50.0))
    }

    var leaderboardScore: Int {
        bestStreak * 1000 + max(0, 999 - durationSeconds)
    }

    var difficulty: Int {
        min(10, max(1, bestStreak / 5 + 1))
    }

    var accuracy: Double {
        guard totalSeen > 0 else { return 0 }
        return Double(totalCorrect) / Double(totalSeen)
    }

    var ratingText: String {
        switch bestStreak {
        case 0...5: return "Getting Started"
        case 6...15: return "Not Bad"
        case 16...25: return "Average"
        case 26...40: return "Good"
        case 41...60: return "Great"
        case 61...80: return "Excellent"
        case 81...100: return "Outstanding"
        default: return "Legendary"
        }
    }

    func startGame() {
        phase = .playing
        startTime = Date()
        lives = 3
        streak = 0
        bestStreak = 0
        totalSeen = 0
        totalCorrect = 0
        seenWords = []
        wordPool = wordBank.shuffled()

        if let seed = challengeSeed {
            rng = SeededGenerator(seed: UInt64(seed))
            wordPool = wordBank.shuffled(using: &rng!)
        }

        presentNextWord()
    }

    func answerSeen() {
        guard !showingFeedback else { return }
        totalSeen += 1

        if !currentWordIsNew {
            // Correct — this word WAS seen before
            totalCorrect += 1
            streak += 1
            bestStreak = max(bestStreak, streak)
            HapticService.shared.correct()
        } else {
            // Wrong — this word was new
            streak = 0
            lives -= 1
            HapticService.shared.wrong()
            if lives <= 0 {
                withAnimation(.easeInOut(duration: 0.3)) {
                    phase = .finished
                }
                return
            }
        }
        presentNextWord()
    }

    func answerNew() {
        guard !showingFeedback else { return }
        totalSeen += 1

        if currentWordIsNew {
            // Correct — this word IS new
            totalCorrect += 1
            streak += 1
            bestStreak = max(bestStreak, streak)
            HapticService.shared.correct()
            seenWords.insert(currentWord)
        } else {
            // Wrong — this word was seen before
            streak = 0
            lives -= 1
            HapticService.shared.wrong()
            if lives <= 0 {
                withAnimation(.easeInOut(duration: 0.3)) {
                    phase = .finished
                }
                return
            }
        }
        presentNextWord()
    }

    private func presentNextWord() {
        // Probability of showing a seen word increases as pool grows
        // Start with 30% chance, scale up to 50%
        let seenChance = seenWords.count >= 3 ? min(0.50, 0.30 + Double(seenWords.count) * 0.01) : 0.0
        let roll: Double
        if var r = rng {
            roll = Double.random(in: 0...1, using: &r)
            rng = r
        } else {
            roll = Double.random(in: 0...1)
        }

        if roll < seenChance, !seenWords.isEmpty {
            // Show a previously seen word
            if var r = rng {
                currentWord = Array(seenWords).randomElement(using: &r) ?? ""
                rng = r
            } else {
                currentWord = Array(seenWords).randomElement() ?? ""
            }
            currentWordIsNew = false
        } else {
            // Show a new word
            if wordPool.isEmpty {
                wordPool = wordBank.filter { !seenWords.contains($0) }.shuffled()
            }
            if let word = wordPool.popLast() {
                currentWord = word
                currentWordIsNew = true
            }
        }
    }

    func reset() {
        phase = .setup
        startTime = nil
        lives = 3
        streak = 0
        bestStreak = 0
        totalSeen = 0
        totalCorrect = 0
        seenWords = []
        wordPool = []
        currentWord = ""
    }
}

    // MARK: - Save & Share

    private func saveExercise() {
        guard !exerciseSaved else { return }
        exerciseSaved = true
        paywallTrigger.recordExerciseCompleted()
        trainingManager.addTrainingTime(viewModel.durationSeconds)

        let exercise = Exercise(
            type: .verbalMemory,
            difficulty: viewModel.difficulty,
            score: viewModel.score,
            durationSeconds: viewModel.durationSeconds
        )
        modelContext.insert(exercise)

        let descriptor = FetchDescriptor<DailySession>(sortBy: [SortDescriptor(\.date, order: .reverse)])
        let allSessions = (try? modelContext.fetch(descriptor)) ?? []
        let session: DailySession
        if let existing = allSessions.first(where: { Calendar.current.isDateInToday($0.date) }) {
            session = existing
        } else {
            session = DailySession()
            modelContext.insert(session)
        }
        session.addExercise(exercise)
        user?.updateStreak()

        isNewPersonalBest = PersonalBestTracker.shared.record(score: viewModel.leaderboardScore, for: .verbalMemory)

        if let user {
            _ = ContentView.awardXP(
                user: user,
                score: viewModel.score,
                difficulty: viewModel.difficulty,
                achievementService: achievementService,
                modelContext: modelContext,
                gameCenterService: gameCenterService,
                exerciseType: .verbalMemory,
                gameScore: viewModel.bestStreak
            )
        }

        gameCenterService.reportScore(viewModel.leaderboardScore, leaderboardID: GameCenterService.verbalMemoryLeaderboard)
    }

    private func generateShareCard() {
        let card = ExerciseShareCard(
            exerciseName: "Verbal Memory",
            exerciseIcon: "text.book.closed.fill",
            accentColor: AppColors.violet,
            mainValue: "\(viewModel.bestStreak)",
            mainLabel: "BEST STREAK",
            ratingText: viewModel.ratingText,
            stats: [
                ("Accuracy", String(format: "%.0f%%", viewModel.accuracy * 100)),
                ("Words", "\(viewModel.totalSeen)")
            ]
        )
        shareImage = card.renderAsImage(size: CGSize(width: 360, height: 640), scale: 3)
    }
