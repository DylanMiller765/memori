> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add shareable friend challenges via URL-encoded deep links for all 10 games, a real Game Center daily challenge leaderboard, and remove all simulated/mock leaderboard data.

```swift
import Foundation

struct ChallengeLink: Equatable {
    let game: ExerciseType
    let seed: Int
    let score: Int
    let challengerName: String

    /// Generate a shareable URL
    var url: URL? {
        var components = URLComponents()
        components.scheme = "memori"
        components.host = "duel"
        components.queryItems = [
            URLQueryItem(name: "game", value: game.rawValue),
            URLQueryItem(name: "seed", value: "\(seed)"),
            URLQueryItem(name: "score", value: "\(score)"),
            URLQueryItem(name: "name", value: challengerName),
        ]
        return components.url
    }

    /// Parse from incoming URL
    init?(url: URL) {
        guard url.scheme == "memori", url.host == "duel" else { return nil }
        let params = URLComponents(url: url, resolvingAgainstBaseURL: false)?.queryItems
        guard let gameRaw = params?.first(where: { $0.name == "game" })?.value,
              let game = ExerciseType(rawValue: gameRaw),
              let seedStr = params?.first(where: { $0.name == "seed" })?.value,
              let seed = Int(seedStr),
              let scoreStr = params?.first(where: { $0.name == "score" })?.value,
              let score = Int(scoreStr),
              let name = params?.first(where: { $0.name == "name" })?.value
        else { return nil }

        self.game = game
        self.seed = seed
        self.score = score
        self.challengerName = name
    }

    init(game: ExerciseType, seed: Int, score: Int, challengerName: String) {
        self.game = game
        self.seed = seed
        self.score = score
        self.challengerName = challengerName
    }

    /// Generate a random seed for a new challenge
    static func randomSeed() -> Int {
        Int.random(in: 10000...99999)
    }
}
```

### Task 2: Extend DeepLinkRouter for challenges

**Files:**
- Modify: `MindRestore/Services/DeepLinkRouter.swift`

- [ ] **Step 1: Add challenge destination and pending challenge state**

Add `.challenge(ChallengeLink)` to `DeepLinkDestination` enum. Add `pendingChallenge: ChallengeLink?` property. Handle `"duel"` host in the URL handler by parsing query params into ChallengeLink.

```swift
enum DeepLinkDestination: Equatable {
    case home
    case train
    case game(ExerciseType)
    case compete
    case insights
    case profile
    case dailyChallenge
    case challenge(ChallengeLink)
}

@MainActor @Observable
final class DeepLinkRouter {
    var pendingDestination: DeepLinkDestination?
    var pendingChallenge: ChallengeLink?

    func handle(_ url: URL) {
        guard url.scheme == "memori" else { return }

        switch url.host {
        case "home": pendingDestination = .home
        case "train": pendingDestination = .train
        case "compete": pendingDestination = .compete
        case "insights": pendingDestination = .insights
        case "profile": pendingDestination = .profile
        case "challenge": pendingDestination = .dailyChallenge
        case "duel":
            if let link = ChallengeLink(url: url) {
                pendingChallenge = link
                pendingDestination = .challenge(link)
            } else {
                pendingDestination = .train
            }
        case "game":
            if let typeName = url.pathComponents.dropFirst().first,
               let type = ExerciseType(rawValue: typeName) {
                pendingDestination = .game(type)
            } else {
                pendingDestination = .train
            }
        default:
            pendingDestination = .home
        }
    }
}
```

struct SeededGenerator: RandomNumberGenerator {

### Task 7: Create ChallengeResultView

**Files:**
- Create: `MindRestore/Views/Social/ChallengeResultView.swift`

- [ ] **Step 1: Build side-by-side comparison view**

Shown after Player B finishes a challenged game. Displays both scores, who won.

```swift
struct ChallengeResultView: View {
    let challenge: ChallengeLink  // Original challenge (challenger's score)
    let playerScore: Int          // Player B's score
    let playerName: String        // Player B's name (from Game Center or "You")
    let onShareResult: () -> Void
    let onChallengeAnother: () -> Void
    let onDone: () -> Void

    private var playerWon: Bool {
        // For reaction time, lower is better
        if challenge.game == .reactionTime {
            return playerScore < challenge.score
        }
        return playerScore > challenge.score
    }

    var body: some View {
        VStack(spacing: 20) {
            Text(playerWon ? "You Won!" : "They Got You!")
                .font(.title.weight(.bold))
                .foregroundStyle(playerWon ? AppColors.mint : AppColors.coral)

            // Side-by-side scores
            HStack(spacing: 24) {
                // Challenger
                VStack(spacing: 8) {
                    Text(challenge.challengerName)
                        .font(.subheadline.weight(.semibold))
                    Text("\(challenge.score)")
                        .font(.system(size: 32, weight: .bold, design: .rounded))
                        .foregroundStyle(!playerWon ? AppColors.mint : .secondary)
                }
                .frame(maxWidth: .infinity)

                Text("vs")
                    .font(.headline)
                    .foregroundStyle(.secondary)

                // Player
                VStack(spacing: 8) {
                    Text("You")
                        .font(.subheadline.weight(.semibold))
                    Text("\(playerScore)")
                        .font(.system(size: 32, weight: .bold, design: .rounded))
                        .foregroundStyle(playerWon ? AppColors.mint : .secondary)
                }
                .frame(maxWidth: .infinity)
            }
            .appCard()
            .padding(.horizontal)

            // Actions
            Button { onShareResult() } label: {
                HStack { Image(systemName: "square.and.arrow.up"); Text("Share Result") }
                    .accentButton()
            }
            .padding(.horizontal, 32)

            Button { onChallengeAnother() } label: {
                Text("Challenge Someone Else")
                    .gradientButton()
            }
            .padding(.horizontal, 32)

            Button { onDone() } label: {
                Text("Done").font(.headline).foregroundStyle(.secondary)
            }
        }
    }
}
```
